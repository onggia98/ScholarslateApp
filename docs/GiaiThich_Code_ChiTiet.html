V1 — Enable Extensions

db/migration/V1__enable_extensions.sql

Script đầu tiên Flyway chạy — bật pgvector extension. Nếu thiếu, cột vector(384) trong V2 sẽ fail với lỗi "type vector does not exist".

SQL
-- Idempotent: IF NOT EXISTS đảm bảo chạy lại không lỗi
CREATE EXTENSION IF NOT EXISTS vector;

-- pg_trgm: ILIKE fuzzy search — không dùng trong dự án này nhưng để sẵn
-- CREATE EXTENSION IF NOT EXISTS pg_trgm;
Tại sao phải là V1? Flyway chạy các script theo thứ tự version. V2 tạo bảng với cột embedding vector(384) — PostgreSQL cần biết type vector trước. Nếu đảo thứ tự → migration fail ngay tại CREATE TABLE.

V3 — Constraints

db/migration/V3__create_constraints.sql

Thêm UNIQUE, FOREIGN KEY và CHECK constraints — đây là tầng bảo vệ data integrity ở DB level, độc lập với Java code.

SQL
-- UNIQUE: ngăn email đăng ký 2 lần
ALTER TABLE "user" ADD CONSTRAINT uq_user_email UNIQUE (email);

-- UNIQUE: topic name duy nhất trong cùng 1 user
ALTER TABLE topic ADD CONSTRAINT uq_topic_user_name UNIQUE (user_id, name);

-- UNIQUE: 1 paper không được lưu 2 lần (idempotency)
ALTER TABLE paper ADD CONSTRAINT uq_paper_arxiv_id UNIQUE (arxiv_id);

-- UNIQUE: 1 user chỉ favorite 1 paper 1 lần
-- PostgreSQL tự tạo B-tree index → không cần tạo thêm index riêng
ALTER TABLE favorite ADD CONSTRAINT uq_favorite UNIQUE (user_id, paper_id);

-- UNIQUE: 1 (user, paper, type) chỉ có 1 notification
ALTER TABLE notification ADD CONSTRAINT uq_notification UNIQUE (user_id, paper_id, type);

-- FK: topic → user ON DELETE CASCADE (user bị xóa → topic bị xóa)
ALTER TABLE topic ADD CONSTRAINT fk_topic_user
    FOREIGN KEY (user_id) REFERENCES "user"(id) ON DELETE CASCADE;

-- FK self-referential: paper → paper (original_paper_id)
-- ON DELETE SET NULL: paper gốc bị xóa → duplicate paper tự NULL
ALTER TABLE paper ADD CONSTRAINT fk_paper_original
    FOREIGN KEY (original_paper_id) REFERENCES paper(id) ON DELETE SET NULL;

-- FK: paper_topic → topic ON DELETE CASCADE
-- Xóa topic → xóa liên kết paper_topic, NHƯNG paper vẫn tồn tại
ALTER TABLE paper_topic ADD CONSTRAINT fk_pt_topic
    FOREIGN KEY (topic_id) REFERENCES topic(id) ON DELETE CASCADE;

-- CHECK: role chỉ nhận 2 giá trị
ALTER TABLE "user" ADD CONSTRAINT chk_user_role
    CHECK (role IN ('USER', 'ADMIN'));

-- CHECK: processing_status 3 trạng thái hợp lệ
ALTER TABLE paper ADD CONSTRAINT chk_processing_status
    CHECK (processing_status IN ('PENDING', 'DONE', 'FAILED'));

-- CHECK: quality_score NULL hoặc trong [0.0, 10.0]
ALTER TABLE paper ADD CONSTRAINT chk_quality_score
    CHECK (quality_score IS NULL OR (quality_score >= 0.0 AND quality_score <= 10.0));
DB-level vs Application-level validation Java code validate business rules (max 10 topic/user, max 5 keyword/topic). DB constraints validate data integrity (unique email, valid status enum). Cả hai cùng tồn tại — DB là safety net cuối cùng nếu Java bị bypass.

V4 — Indexes

db/migration/V4__create_indexes.sql

Đây là migration quan trọng nhất về performance. Thiếu index → full table scan → query chậm hàng giây thay vì milliseconds.

SQL
-- ── HNSW VECTOR INDEX — bắt buộc Day 1 ────────────────────────────────────
-- Hierarchical Navigable Small World: approximate nearest neighbor graph
-- m=16: số neighbor mỗi node (accuracy vs memory tradeoff)
-- ef_construction=64: beam width khi build index (accuracy vs build time)
-- vector_cosine_ops: dùng cosine distance (phù hợp sentence embeddings)
CREATE INDEX idx_paper_embedding_hnsw
    ON paper
    USING hnsw (embedding vector_cosine_ops)
    WITH (m = 16, ef_construction = 64);

-- ef_search=128 set GLOBAL qua HikariCP connection-init-sql
-- → không cần SET LOCAL trong từng query
-- Cao hơn ef_construction (64) → recall tốt hơn khi query

-- ── GIN FULL-TEXT: 2 INDEX RIÊNG BIỆT ─────────────────────────────────────
-- QUAN TRỌNG: expression trong CREATE INDEX phải KHỚP HOÀN TOÀN
-- với expression trong WHERE clause mới dùng được functional index

-- Index 1 (UC05): title + abstract + authors → plainto_tsquery (tìm tên tác giả)
CREATE INDEX idx_paper_fts_search ON paper
    USING gin (
        to_tsvector('english',
            coalesce(title,'') || ' ' ||
            coalesce(abstract,'') || ' ' ||
            coalesce(authors,''))
    );

-- Index 2 (UC12): title + abstract ONLY → phraseto_tsquery (exact phrase)
-- Bỏ authors vì keyword topic là chủ đề, không phải tên người
-- Nếu include authors → query không dùng được index này → seq scan
CREATE INDEX idx_paper_fts_topic ON paper
    USING gin (
        to_tsvector('english',
            coalesce(title,'') || ' ' ||
            coalesce(abstract,''))
    );

-- ── B-TREE INDEXES ─────────────────────────────────────────────────────────
CREATE INDEX idx_paper_published_at  ON paper(published_at);  -- sort
CREATE INDEX idx_paper_status        ON paper(processing_status); -- retry query
CREATE INDEX idx_paper_is_duplicate  ON paper(is_duplicate);  -- filter
CREATE INDEX idx_topic_user_active   ON topic(user_id, is_active); -- scheduler
CREATE INDEX idx_notification_user_read ON notification(user_id, is_read, created_at);

-- FAVORITE: UNIQUE constraint trong V3 đã tạo index tự động → không khai báo thêm

@Entity @Table(name = "topic")
public class Topic {
    @Id @GeneratedValue(strategy = GenerationType.UUID)
    private UUID id;

    @Column(nullable = false, length = 255)
    private String name;

    /**
     * Comma-separated, max 5 keywords, tối đa 255 ký tự.
     * Ví dụ: "large language model,rag,vector database"
     * Format kiểm soát tại Service.normalizeKeywords()
     * Không dùng @ElementCollection vì keywords là static list nhỏ
     */
    @Column(length = 255)
    private String keywords;

    @Column(name = "is_active", nullable = false)
    @Builder.Default
    private Boolean isActive = true;

    // LAZY: không cần load User khi chỉ cần Topic info
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "user_id", nullable = false)
    private User user;

    @CreationTimestamp @UpdateTimestamp
    private OffsetDateTime createdAt, updatedAt;
}
Tại sao keywords lưu dạng comma-separated thay vì bảng riêng? Max 5 keywords/topic, không cần query riêng từng keyword. Lưu dạng VARCHAR(255) đơn giản hơn nhiều — tránh join phức tạp. Topic matching dùng Java split(",") rồi từng keyword gọi matchesTopic().

@Entity @Table(name = "paper")
@Getter @Setter @NoArgsConstructor @AllArgsConstructor @Builder
public class Paper {

    @Id @GeneratedValue(strategy = GenerationType.UUID)
    private UUID id;

    @Column(name = "arxiv_id", unique = true, length = 50)
    private String arxivId;  // e.g. "2301.00001" — không có version suffix

    // "abstract" là Java keyword → dùng abstractText làm field name
    // @Column(name="abstract") map sang đúng tên cột DB
    @Column(name = "abstract", columnDefinition = "TEXT")
    private String abstractText;

    /**
     * vector(384) — 384 floats × 4 bytes = ~1.5KB per row.
     * Không thể dùng AttributeConverter: setString() gây type mismatch.
     * Phải dùng VectorUserType (PGobject.type="vector") để JDBC
     * gửi đúng OID cho PostgreSQL JDBC driver.
     */
    @Type(VectorUserType.class)
    @Column(columnDefinition = "vector(384)")
    private float[] embedding;

    @Column(name = "processing_status", nullable = false)
    @Builder.Default
    private String processingStatus = "PENDING";

    @Column(name = "retry_count")
    @Builder.Default
    private Integer retryCount = 0;

    // Self-referential: paper trùng lặp trỏ về paper gốc
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "original_paper_id")
    private Paper originalPaper;

    // LAZY: tránh N+1 khi list papers — topics chỉ cần khi xem chi tiết
    @ManyToMany(fetch = FetchType.LAZY)
    @JoinTable(name = "paper_topic",
        joinColumns = @JoinColumn(name = "paper_id"),
        inverseJoinColumns = @JoinColumn(name = "topic_id"))
    @Builder.Default
    private List<Topic> topics = new ArrayList<>();

    @CreationTimestamp @UpdateTimestamp
    private OffsetDateTime createdAt, updatedAt;
}
FieldChi tiết thiết kế
arxivId uniquePrimary key nghiệp vụ — nếu scheduler fetch paper đã có → ON CONFLICT DO NOTHING bỏ qua
abstractText → cột "abstract"abstract là keyword Java (abstract class). Đặt tên field abstractText, dùng @Column(name="abstract") để map đúng cột DB. DTO dùng @JsonProperty("abstract") để JSON field vẫn là "abstract"
float[] primitive arrayKhông dùng Float[] (boxed) — tiết kiệm memory, tránh boxing overhead khi tính cosine distance
@Builder.Default trên topicsKhông có @Builder.Default → Paper.builder().build().getTopics() trả null → NullPointerException khi stream. Default new ArrayList<>() an toàn hơn
FetchType.LAZY cho topicsMỗi paper có thể liên kết nhiều topics. EAGER sẽ JOIN ngay khi load paper — với 50 papers/page → 50 JOIN queries. LAZY chỉ load khi cần (trong @Transactional context)

Entity — PaperTopic + PaperTopicId

entity/PaperTopic.java & PaperTopicId.java

Bảng liên kết N-N giữa Paper và Topic. Có PK tổng hợp (composite primary key) — không có cột id riêng.

Java
// PaperTopicId — Composite PK, bắt buộc implements Serializable
@Embeddable
@Getter @Setter @NoArgsConstructor @AllArgsConstructor @EqualsAndHashCode
public class PaperTopicId implements Serializable {
    @Column(name = "paper_id") private UUID paperId;
    @Column(name = "topic_id") private UUID topicId;
}

// PaperTopic — Entity với composite PK
@Entity @Table(name = "paper_topic")
public class PaperTopic {

    @EmbeddedId
    private PaperTopicId id;

    // @MapsId("paperId"): map field paperId trong @EmbeddedId với FK paper_id
    @ManyToOne(fetch = FetchType.LAZY)
    @MapsId("paperId")
    @JoinColumn(name = "paper_id")
    private Paper paper;

    @ManyToOne(fetch = FetchType.LAZY)
    @MapsId("topicId")
    @JoinColumn(name = "topic_id")
    private Topic topic;

    @CreationTimestamp
    private OffsetDateTime createdAt;
}
Tại sao cần Serializable cho @EmbeddedId? JPA spec yêu cầu composite PK class phải implements Serializable — Hibernate cần serialize PK để lưu vào 2nd-level cache và để so sánh entity identity. Thiếu Serializable → runtime error khi persist.

@Component
public class JwtUtil {
    private final SecretKey secretKey;
    private final long expirationMs;  // 86400000 = 24 giờ

    public JwtUtil(@Value("${jwt.secret}") String secret,
                   @Value("${jwt.expiration-ms}") long expirationMs) {
        // Keys.hmacShaKeyFor: yêu cầu secret >= 256 bit (32 bytes)
        // jwt.secret không có default value → app fail-fast nếu không set
        this.secretKey = Keys.hmacShaKeyFor(secret.getBytes(UTF_8));
        this.expirationMs = expirationMs;
    }

    public String generateToken(UUID userId, String email, String role) {
        Date now = new Date();
        return Jwts.builder()
            .subject(userId.toString())   // sub claim = userId
            .claim("email", email)        // frontend decode để hiển thị
            .claim("role", role)          // "USER" | "ADMIN"
            .issuedAt(now)
            .expiration(new Date(now.getTime() + expirationMs))
            .signWith(secretKey)           // HMAC-SHA256 auto-detect từ key type
            .compact();
    }

    public UUID extractUserId(String token) {
        return UUID.fromString(extractClaims(token).getSubject());
    }

    public boolean isTokenValid(String token) {
        try {
            // Vừa parse vừa verify signature — nếu sai signature → JwtException
            return extractClaims(token).getExpiration().after(new Date());
        } catch (JwtException | IllegalArgumentException e) {
            return false;  // giả mạo, hết hạn, hoặc format sai
        }
    }

    private Claims extractClaims(String token) {
        return Jwts.parser()
            .verifyWith(secretKey)
            .build()
            .parseSignedClaims(token)
            .getPayload();
    }
}

// OncePerRequestFilter: đảm bảo filter chạy đúng 1 lần mỗi request
@Component @RequiredArgsConstructor
public class JwtAuthenticationFilter extends OncePerRequestFilter {

    @Override
    protected void doFilterInternal(HttpServletRequest request,
                                     HttpServletResponse response,
                                     FilterChain filterChain) {
        String token = extractToken(request); // "Authorization: Bearer <token>"

        if (StringUtils.hasText(token) && jwtUtil.isTokenValid(token)) {
            UUID userId = jwtUtil.extractUserId(token);
            String role  = jwtUtil.extractRole(token);

            // principal = UUID userId (không phải UserDetails)
            // Controller lấy qua @AuthenticationPrincipal UUID userId
            var auth = new UsernamePasswordAuthenticationToken(
                userId, null,
                List.of(new SimpleGrantedAuthority("ROLE_" + role))
            );
            auth.setDetails(new WebAuthenticationDetailsSource().buildDetails(request));
            SecurityContextHolder.getContext().setAuthentication(auth);
        }

        // Không throw exception — request không có token vẫn đi tiếp
        // SecurityFilterChain xử lý việc từ chối nếu endpoint yêu cầu auth
        filterChain.doFilter(request, response);
    }

    private String extractToken(HttpServletRequest request) {
        String header = request.getHeader("Authorization");
        if (StringUtils.hasText(header) && header.startsWith("Bearer "))
            return header.substring(7);  // bỏ "Bearer " (7 ký tự)
        return null;
    }
}

public class VectorUserType implements UserType<float[]> {

    @Override
    public int getSqlType() { return Types.OTHER; }
    // pgvector là custom PostgreSQL type → không có JDBC standard type
    // Types.OTHER → JDBC driver dùng setObject() thay vì setString()

    @Override
    public float[] nullSafeGet(ResultSet rs, int position, ...) throws SQLException {
        String value = rs.getString(position);  // DB trả "[0.1,0.2,...,0.384]"
        return value == null ? null : parseVector(value);
    }

    @Override
    public void nullSafeSet(PreparedStatement st, float[] value, int index, ...) {
        if (value == null) {
            st.setNull(index, Types.OTHER);
        } else {
            PGobject pgObj = new PGobject();
            pgObj.setType("vector");           // ← KEY: PostgreSQL nhận đúng OID
            pgObj.setValue(toVectorString(value)); // float[] → "[0.1,...,0.384]"
            st.setObject(index, pgObj);
        }
    }

    private String toVectorString(float[] arr) {
        StringBuilder sb = new StringBuilder("[");
        for (int i = 0; i < arr.length; i++) {
            if (i > 0) sb.append(",");
            sb.append(arr[i]);
        }
        return sb.append("]").toString();
    }

    private float[] parseVector(String s) {
        // "[0.1,0.2,...,0.384]" → strip brackets → split → parse
        String trimmed = s.trim();
        if (trimmed.startsWith("[")) trimmed = trimmed.substring(1, trimmed.length() - 1);
        String[] parts = trimmed.split(",");
        float[] result = new float[parts.length];
        for (int i = 0; i < parts.length; i++)
            result[i] = Float.parseFloat(parts[i].trim());
        return result;
    }

    @Override public float[] deepCopy(float[] v) { return v == null ? null : Arrays.copyOf(v,v.length); }
    @Override public boolean isMutable() { return true; }  // float[] là mutable
}
Tại sao KHÔNG dùng AttributeConverter? AttributeConverter<float[], String> dùng PreparedStatement.setString() → PostgreSQL nhận parameter là character varying (VARCHAR). Nhưng cột kiểu vector(384) cần implicit cast từ text sang vector → lỗi: "column is of type vector but expression is of type character varying".      */
    private String normalizeKeywords(String keywords) {
        if (keywords == null || keywords.isBlank()) return null;
        List<String> kws = Arrays.stream(keywords.split(","))
            .map(String::trim).map(String::toLowerCase)
            .filter(k -> !k.isEmpty()).distinct().toList();
        if (kws.size() > maxKeywordsPerTopic)
            throw AppException.badRequest("Maximum " + maxKeywordsPerTopic + " keywords");
        return String.join(",", kws);
    }
}

private float[] meanPool(List<List<Number>> tokenEmbeddings) {
    // shape: [seq_len, hidden_size] → average over seq_len → [hidden_size]
    int dims = tokenEmbeddings.get(0).size();
    float[] result = new float[dims];
    for (var token : tokenEmbeddings)
        for (int d = 0; d < dims; d++) result[d] += token.get(d).floatValue();
    for (int d = 0; d < dims; d++) result[d] /= tokenEmbeddings.size();
    return result;
}

// REQUIRES_NEW: mỗi paper = transaction riêng
// Lỗi paper 5 không rollback paper 1-4 đã COMMIT
@Transactional(propagation = Propagation.REQUIRES_NEW)
public boolean process(Paper paper) {
    if (paperRepository.existsByArxivId(paper.getArxivId())) return false; // skip duplicate
    Paper saved = paperRepository.save(paper); // cần ID trước khi query vector
    runPipeline(saved);
    return true;
}

@Transactional(propagation = Propagation.REQUIRES_NEW)
public void retry(Paper paper) {
    paper.setRetryCount(paper.getRetryCount() + 1);
    paper.setLastRetryAt(OffsetDateTime.now());
    paper.setProcessingStatus("PENDING");
    paper.setLastError(null);
    paperRepository.save(paper);
    runPipeline(paper);
}

private void runPipeline(Paper paper) {
    // Step 1: Embedding
    float[] embedding = embeddingService.embed(paper.getTitle() + " " + paper.getAbstractText());
    if (embedding == null) { markFailed(paper, "Embedding failed"); return; }
    paper.setEmbedding(embedding);

    // Step 2: Duplicate detection
    String embStr = VectorUtils.toVectorString(embedding);
    var dup = paperRepository.findDuplicate(
        paper.getId(), embStr, OffsetDateTime.now().minusDays(90));
    if (dup.isPresent()) {
        paper.setIsDuplicate(true);
        paper.setOriginalPaper(dup.get());
        paper.setProcessingStatus("DONE");
        paperRepository.save(paper);
        return;  // duplicate không cần Groq hay topic matching
    }

    // Step 3: Groq summary + quality_score
    var result = groqService.summarize(paper.getTitle(), paper.getAbstractText());
    if (!result.isValid()) { markFailed(paper, result.error()); return; }
    paper.setSummary(result.summary());
    paper.setQualityScore(result.qualityScore());

    // Step 4: Topic matching + notification
    for (Topic topic : topicRepository.findByIsActiveTrue()) {
        boolean matched = Arrays.stream(topic.getKeywords().split(","))
            .map(String::trim)
            .anyMatch(kw -> paperRepository.matchesTopic(paper.getId(), kw));
        if (matched) {
            paperTopicRepository.insertIfNotExists(paper.getId(), topic.getId());
            notificationRepository.insertIfNotExists(
                topic.getUser().getId(), paper.getId(),
                "New paper matching \\" + topic.getName() + "\\": " + paper.getTitle());
        }
    }

    // Step 5: Mark DONE
    paper.setProcessingStatus("DONE");
    paperRepository.save(paper);
}

private void markFailed(Paper paper, String error) {
    paper.setProcessingStatus("FAILED");
    paper.setLastError(error);
    paperRepository.save(paper);
    log.warn("Pipeline FAILED — arxivId={}: {}", paper.getArxivId(), error);
}

@Scheduled(cron = "${scheduler.main.cron:0 0 6 * * *}")
public void runDailyFetch() {
    var activeTopics = topicRepository.findByIsActiveTrue();
    if (activeTopics.isEmpty()) { log.info("No active topics, skip"); return; }

    // LinkedHashSet: giữ insertion order + dedup
    // 2 users có topic "machine learning" → fetch keyword đó 1 lần duy nhất
    Set<String> uniqueKeywords = new LinkedHashSet<>();
    for (var topic : activeTopics)
        Arrays.stream(topic.getKeywords().split(","))
            .map(String::trim).filter(k -> !k.isEmpty())
            .forEach(uniqueKeywords::add);

    for (String keyword : uniqueKeywords) {
        var papers = arxivFetchService.fetchByKeyword(keyword);
        for (var paper : papers) {
            try {
                // REQUIRES_NEW transaction: exception 1 paper không stop cả batch
                paperPipelineService.process(paper);
            } catch (Exception e) {
                log.error("Pipeline error for {}: {}", paper.getArxivId(), e.getMessage());
            }
        }
    }
}

// RetryScheduler: mỗi 30 phút SAU KHI batch trước kết thúc
@Scheduled(fixedDelayString = "${scheduler.retry.fixed-delay-ms:1800000}")
public void retryFailed() {
    // Query: FAILED AND retryCount < 3
    var failed = paperRepository.findFailedPapersForRetry(3);
    if (failed.isEmpty()) return;
    log.info("RetryScheduler: retrying {} papers", failed.size());
    for (var paper : failed) {

// Decode JWT payload mà không cần thư viện ngoài
export function decodeJwt(token: string): JwtPayload | null {
  const parts = token.split('.');          // header.payload.signature
  if (parts.length < 2) return null;

  let b64 = parts[1]
    .replace(/-/g, '+')                  // base64url → base64 standard
    .replace(/_/g, '/');
  while (b64.length % 4) b64 += '=';   // add padding

  const json = decodeURIComponent(
    atob(b64).split('')
      .map(c => '%' + ('00' + c.charCodeAt(0).toString(16)).slice(-2))
      .join('')
  );
  return JSON.parse(json) as JwtPayload;
}

// Relative time: "2m ago", "3h ago", "5d ago"
export function timeAgo(iso: string): string {
  const s = Math.max(1, Math.round((Date.now() - new Date(iso).getTime()) / 1000));
  if (s < 60)  return s + 's ago';
  if (s < 3600) return Math.round(s/60) + 'm ago';
  if (s < 86400) return Math.round(s/3600) + 'h ago';
  return Math.round(s/86400) + 'd ago';
}

// authorString: "Alice, Bob, Charlie, Dave" → "Alice, Bob, Charlie +1"
export function authorString(authors: string | string[]): string {
  const parts = Array.isArray(authors)
    ? authors
    : String(authors).split(',').map(s => s.trim()).filter(Boolean);
  if (parts.length <= 3) return parts.join(', ');
  return parts.slice(0, 3).join(', ') + ' +' + (parts.length - 3);
}

async function request<T>(path: string, options: RequestInit = {}): Promise<T> {
  const token = getStoredToken();
  const res = await fetch(API_BASE + path, {
    ...options,
    headers: {
      'Content-Type': 'application/json',
      ...(token ? { Authorization: 'Bearer ' + token } : {}),
    },
  });

  if (res.status === 401) throw new ApiError(401, 'Unauthorized');
  if (res.status === 204) return undefined as T;  // DELETE returns 204 No Content

  const body = await res.json();
  if (!res.ok) throw new ApiError(res.status, body?.message);

  // Unwrap ApiResponse<T>: { success, message, data: T } → return data
  if (body !== null && 'data' in body) return body.data as T;
  return body as T;  // fallback nếu backend trả raw JSON
}
